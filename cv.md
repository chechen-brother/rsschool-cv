async function getDomains(links) {
    let domains = [];
    for(item in links) {
        links[item].forEach(i => {
            domains.push(i.split('/').filter(el => el.length > 0)[1]);
        })
    };
    let domainsSet = new Set(domains);
    domains = Array.from(domainsSet);
    return domains;
}
