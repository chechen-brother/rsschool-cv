function findMissingLatters(letters) {
    let alphabet = [];
    for (let i = 97; i <= letters[letters.length - 1].charCodeAt(); i++) {
        alphabet.push(String.fromCharCode(i));
    }
    return alphabet.filter((el) => {
        if (letters.includes(el))
            return false;
        return true;
    });
}
